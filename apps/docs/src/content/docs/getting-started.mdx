# Getting Started

Welcome to **next-unified-query**, a powerful TypeScript library for unified query management with Next.js.

## Installation

Install the package using your preferred package manager:

```bash
npm install next-unified-query
# or
pnpm add next-unified-query
# or
yarn add next-unified-query
```

## Quick Start

Here's a simple example to get you started:

```typescript
import { useQuery } from 'next-unified-query';

function MyComponent() {
  const { data, loading, error } = useQuery('/api/users');
  
  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;
  
  return (
    <div>
      {data.map(user => (
        <div key={user.id}>{user.name}</div>
      ))}
    </div>
  );
}
```

## Configuration

You can configure the library in your `next.config.js`:

```javascript
module.exports = {
  // ... other config
  experimental: {
    unifiedQuery: {
      cache: true,
      retry: 3,
      timeout: 5000
    }
  }
};
```

## Advanced Usage

### Custom Hooks

Create reusable query hooks for your application:

```typescript
import { useQuery } from 'next-unified-query';

export function useUsers() {
  return useQuery('/api/users', {
    staleTime: 60000, // 1 minute
    cacheTime: 300000, // 5 minutes
  });
}

export function useUser(id: string) {
  return useQuery(`/api/users/${id}`, {
    enabled: !!id,
  });
}
```

### Error Handling

Handle errors gracefully with built-in error boundaries:

```typescript
import { useQuery, ErrorBoundary } from 'next-unified-query';

function App() {
  return (
    <ErrorBoundary fallback={<div>Something went wrong</div>}>
      <UserList />
    </ErrorBoundary>
  );
}
```
